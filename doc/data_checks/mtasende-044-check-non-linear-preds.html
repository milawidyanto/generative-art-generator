def points_per_pl(data, verbose=False):
    grouped_datapoints = data.groupBy("partner_id", "locale").agg(f.count("partner_id").alias("counts")) \
                                    .sort(f.col("counts").desc())
    stats = grouped_datapoints.agg(f.avg(f.col("counts")).alias("mean_data_points_per_pl"),
                                   f.min(f.col("counts")).alias("min_data_points_per_pl"),
                                   f.max(f.col("counts")).alias("max_data_points_per_pl"),
                                   f.stddev(f.col("counts")).alias("stddev_data_points_per_pl")
                                  )
    if verbose:
        grouped_datapoints.show()
        print(stats)
    return stats, grouped_datapoints

def num_with_plus_n(data, n=2, verbose=False):
    grouped_pls = data.groupBy("partner_id", "locale").agg(f.count("partner_id").alias("counts"))
    grouped_ps = data.groupBy("partner_id").agg(f.count("partner_id").alias("counts"))
    grouped_ls = data.groupBy("locale").agg(f.count("locale").alias("counts"))
    agg_data = grouped_pls.where(f.col("counts") > n).agg(f.count("counts").alias(f"n_pls_with_plus_{n}")) \
                          .withColumn(f"n_partners_with_plus_{n}", f.lit(grouped_ps.where(f.col("counts") > n).count())) \
                          .withColumn(f"n_locales_with_plus_{n}", f.lit(grouped_ls.where(f.col("counts") > n).count()))
    if verbose:
        print(agg_data)
    return agg_data

def last_cpa_position(data, verbose=False):
    pl_agg = data.groupBy("partner_id", "locale") \
                 .agg(f.max(f.col("filtered_cpa_mean")).alias("max_cpa_mean"),
                      f.min(f.col("filtered_cpa_mean")).alias("min_cpa_mean"),
                      f.avg("last_cpa_r").alias("last_cpa")) \
                 .withColumn("last2max", f.col("last_cpa") - f.col("max_cpa_mean")) \
                 .withColumn("last2min", f.col("last_cpa") - f.col("min_cpa_mean")) \
                 .withColumn("last_is_between", f.when((f.col("last2max") <= 0) & (f.col("last2min") >= 0), 1).otherwise(0)) \
                 .withColumn("last_is_left", f.when(f.col("last2min") < 0, 1).otherwise(0)) \
                 .withColumn("last_is_right", f.when(f.col("last2max") > 0, 1).otherwise(0)) \
                 .withColumn("last_is_outside", f.when((f.col("last2max") > 0) | (f.col("last2min") < 0), 1).otherwise(0))
    
    agg_data = pl_agg.agg(f.sum("last_is_between").alias("num_between"),
                      f.sum("last_is_outside").alias("num_outside"),
                      f.sum("last_is_left").alias("num_left"),
                      f.sum("last_is_right").alias("num_right"),
                      f.avg("last_is_between").alias("ratio_between"),
                      f.avg("last_is_outside").alias("ratio_outside"),
                      f.avg("last_is_left").alias("ratio_left"),
                      f.avg("last_is_right").alias("ratio_right")
                     )
    if verbose:
        print(agg_data)
    return agg_data

def concatenate_results(res):
    result = res['cpa_results'].join(res['bv_share_results']) \
                               .withColumn("n_rows", f.lit(res['n_rows'])) \
                               .join(res["bv_share_inc_pct_results"]) \

                               .withColumn("num_partner_locales", f.lit(res["num_partner_locales"])) \
                               .withColumn("num_partners", f.lit(res["num_partners"])) \
                               .join(res["points_pl_stats"]) \
                               .join(res["points_p_stats"]) \

                               .join(res["x_with_plus_2"]) \
                               .join(res["last_cpa_pos"]) \

                               .withColumn("checked_year_month", f.lit(res["checked_year_month"])) \
                               .withColumn("timestamp", f.current_timestamp())
