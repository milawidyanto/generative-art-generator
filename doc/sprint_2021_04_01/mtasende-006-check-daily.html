def get_count_stats(data, verbose=False):
    counts = data.groupBy("ymd").count()
    stats = counts.agg(f.count("ymd").alias("num_days_with_preds"),
                       f.avg("count").alias("mean_counts"),
                       f.min("count").alias("min_counts"),
                       f.max("count").alias("max_counts"),
                       f.stddev("count").alias("std_counts"))
    if verbose:
        stats.show()
    return stats, counts

def get_date_stats(data, verbose=False):
    dates = data.select("ymd").distinct().sort(f.col("ymd").asc())
    stats = counts.agg(f.count("ymd").alias("num_days"),
                       f.min("ymd").alias("min_ymd"),
                       f.max("ymd").alias("max_ymd"))
    dates.cache()
    if verbose:
        stats.show()
        dates.show()
    return stats, dates

def get_duplicate_stats(data, verbose=False):
    values = data.groupBy("ymd", "cpa_input_sim") \
                     .agg(f.sum("clicks_pred").alias("clicks"), f.sum("bv_pred").alias("bv"))
    
    values.cache()
    stats = duplicates.agg(f.count("ymd").alias("num_days_with_duplicates"),
                           f.avg("count").alias("mean_counts"),
                           f.min("count").alias("min_counts"),
                           f.max("count").alias("max_counts"),
                           f.stddev("count").alias("std_counts"))
    if verbose:
        stats.show()
        duplicates.show()
        
    return stats, duplicates

def get_value_stats(data, verbose=False):
    
    values = data.groupBy("ymd", "cpa_input_sim") \
                 .agg(f.sum("clicks_pred").alias("clicks"),
                      f.sum("bv_pred").alias("bv"))
    
    values.cache()
    stats = values.groupBy("ymd").agg(f.count("cpa_input_sim").alias("num_points"),
                                      f.avg("bv").alias("mean_bv"),
                                      f.min("bv").alias("min_bv"),
                                      f.max("bv").alias("max_bv"),
                                      f.stddev("bv").alias("std_bv"))
    if verbose:
        stats.show()
        values.show()
        
    return stats, values
